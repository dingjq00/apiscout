"""Schema 扫描数据模型"""
from dataclasses import dataclass


@dataclass
class ColumnInfo:
    """字段信息"""
    name: str
    data_type: str              # 原始数据库类型
    normalized_type: str        # 统一类型：integer/string/number/boolean/date-time
    nullable: bool
    default: str | None
    comment: str | None
    max_length: int | None
    numeric_precision: int | None
    is_primary_key: bool
    enum_values: list[dict] | None  # [{"value": "ACTIVE", "count": 500}]

    @property
    def is_enum_candidate(self) -> bool:
        return self.enum_values is not None


@dataclass
class IndexInfo:
    """索引信息"""
    name: str
    columns: list[str]
    is_unique: bool
    is_primary: bool


@dataclass
class ExplicitRelation:
    """数据库中真实存在的外键约束"""
    source_table: str
    source_column: str
    target_table: str
    target_column: str
    constraint_name: str


@dataclass
class InferredRelation:
    """通过命名模式推断的关系"""
    source_table: str
    source_column: str
    target_table: str
    target_column: str
    confidence: float
    evidence: str


@dataclass
class TableInfo:
    """表信息"""
    name: str
    schema: str
    comment: str | None
    row_count: int | None
    columns: list[ColumnInfo]
    indexes: list[IndexInfo]
    sample_rows: list[dict]


@dataclass
class SchemaReport:
    """完整的 Schema 扫描报告"""
    dialect: str
    database: str
    scanned_at: str
    tables: list[TableInfo]
    explicit_relations: list[ExplicitRelation]
    inferred_relations: list[InferredRelation]

    @property
    def total_tables(self) -> int:
        return len(self.tables)

    @property
    def total_columns(self) -> int:
        return sum(len(t.columns) for t in self.tables)

    @property
    def total_foreign_keys(self) -> int:
        return len(self.explicit_relations)

    @property
    def total_inferred_relations(self) -> int:
        return len(self.inferred_relations)

    @property
    def enum_candidates(self) -> int:
        return sum(1 for t in self.tables for c in t.columns if c.is_enum_candidate)
```

# apiscout/core/db_scanner/dialect/__init__.py
"""数据库方言层"""

# apiscout/core/db_scanner/dialect/base.py
"""BaseDialect — 方言抽象接口 + 跨方言类型归一化"""
import re
from abc import ABC, abstractmethod


# 类型归一化映射（正则匹配，顺序重要 — 先匹配更具体的）
_TYPE_RULES: list[tuple[str, str]] = [
    # boolean（必须在 integer 之前，bit 只有 1 位时是 boolean）
    (r"^(bool|boolean|bit)$", "boolean"),
    # integer
    (r"(int|serial|bigserial)", "integer"),
    (r"^NUMBER\(\d+,\s*0\)$", "integer"),
    # number（decimal/float 等带小数的）
    (r"(float|double|real|decimal|numeric|money)", "number"),
    (r"^NUMBER\(\d+,\s*[1-9]\d*\)$", "number"),
    (r"^NUMBER$", "number"),
    # date-time
    (r"(timestamp|datetime|date|time)", "date-time"),
    # json
    (r"^(json|jsonb)$", "object"),
    # binary
    (r"(bytea|blob|varbinary|binary|image|raw)", "binary"),
    # string（兜底之前的特定匹配）
    (r"(varchar|text|char|clob|nvarchar|ntext|nchar|string|uuid|xml)", "string"),
]


def normalize_type(data_type: str) -> str:
    """将数据库原始类型归一化为统一类型

    返回: integer / string / number / boolean / date-time / binary / object
    未知类型默认降级为 string
    """
    dt = data_type.strip()
    for pattern, normalized in _TYPE_RULES:
        if re.search(pattern, dt, re.IGNORECASE):
            return normalized
    return "string"  # 未知类型降级


class BaseDialect(ABC):
    """数据库方言抽象基类 — 每种数据库实现这套接口"""

    name: str = "unknown"

    @abstractmethod
    def get_tables(self, conn) -> list[dict]:
        """获取所有表: [{"name": ..., "schema": ..., "comment": ..., "row_count": ...}]"""

    @abstractmethod
    def get_columns(self, conn, table_name: str, table_schema: str) -> list[dict]:
        """获取表的所有字段: [{"name": ..., "data_type": ..., "nullable": ..., ...}]"""

    @abstractmethod
    def get_foreign_keys(self, conn) -> list[dict]:
        """获取所有外键约束（全局）"""

    @abstractmethod
    def get_indexes(self, conn, table_name: str, table_schema: str) -> list[dict]:
        """获取表的索引"""

    @abstractmethod
    def sample_rows(self, conn, table_name: str, table_schema: str, limit: int = 20) -> list[dict]:
        """随机采样 N 行"""

    @abstractmethod
    def count_distinct(self, conn, table_name: str, table_schema: str, column_name: str) -> int:
        """获取字段的 distinct 值数量"""

    @abstractmethod
    def get_enum_values(self, conn, table_name: str, table_schema: str, column_name: str, limit: int = 50) -> list[dict]:
        """获取枚举字段的值 + 分布: [{"value": ..., "count": ...}]"""
```

# apiscout/core/db_scanner/dialect/postgresql.py
"""PostgreSQL 方言 — information_schema + pg_stat"""
import logging
from apiscout.core.db_scanner.dialect.base import BaseDialect, normalize_type

logger = logging.getLogger(__name__)


class PostgreSQLDialect(BaseDialect):

    name = "postgresql"

    def get_tables(self, conn) -> list[dict]:
        sql = """
        SELECT
            t.table_schema,
            t.table_name,
            pg_catalog.obj_description(c.oid, 'pg_class') AS comment,
            COALESCE(s.n_live_tup, 0) AS row_count
        FROM information_schema.tables t
        LEFT JOIN pg_catalog.pg_class c
            ON c.relname = t.table_name
            AND c.relnamespace = (SELECT oid FROM pg_namespace WHERE nspname = t.table_schema)
        LEFT JOIN pg_stat_user_tables s
            ON s.relname = t.table_name AND s.schemaname = t.table_schema
        WHERE t.table_type = 'BASE TABLE'
            AND t.table_schema NOT IN ('pg_catalog', 'information_schema')
        ORDER BY t.table_schema, t.table_name
        """
        cursor = conn.cursor()
        cursor.execute(sql)
        return [
            {"schema": row[0], "name": row[1], "comment": row[2], "row_count": row[3]}
            for row in cursor.fetchall()
        ]

    def get_columns(self, conn, table_name: str, table_schema: str) -> list[dict]:
        sql = """
        SELECT
            c.column_name,
            c.data_type,
            c.is_nullable,
            c.column_default,
            pg_catalog.col_description(
                (SELECT oid FROM pg_class WHERE relname = c.table_name
                 AND relnamespace = (SELECT oid FROM pg_namespace WHERE nspname = c.table_schema)),
                c.ordinal_position
            ) AS comment,
            c.character_maximum_length,
            c.numeric_precision,
            EXISTS(
                SELECT 1 FROM information_schema.key_column_usage kcu
                JOIN information_schema.table_constraints tc
                    ON kcu.constraint_name = tc.constraint_name
                    AND kcu.table_schema = tc.table_schema
                WHERE tc.constraint_type = 'PRIMARY KEY'
                    AND kcu.table_name = c.table_name
                    AND kcu.table_schema = c.table_schema
                    AND kcu.column_name = c.column_name
            ) AS is_primary_key
        FROM information_schema.columns c
        WHERE c.table_name = %s AND c.table_schema = %s
        ORDER BY c.ordinal_position
        """
        cursor = conn.cursor()
        cursor.execute(sql, (table_name, table_schema))
        return [
            {
                "name": row[0],
                "data_type": row[1],
                "normalized_type": normalize_type(row[1]),
                "nullable": row[2] == "YES",
                "default": row[3],
                "comment": row[4],
                "max_length": row[5],
                "numeric_precision": row[6],
                "is_primary_key": bool(row[7]),
            }
            for row in cursor.fetchall()
        ]

    def get_foreign_keys(self, conn) -> list[dict]:
        sql = """
        SELECT
            tc.constraint_name,
            kcu.table_schema AS source_schema,
            kcu.table_name AS source_table,
            kcu.column_name AS source_column,
            ccu.table_schema AS target_schema,
            ccu.table_name AS target_table,
            ccu.column_name AS target_column
        FROM information_schema.table_constraints tc
        JOIN information_schema.key_column_usage kcu
            ON tc.constraint_name = kcu.constraint_name
            AND tc.table_schema = kcu.table_schema
        JOIN information_schema.constraint_column_usage ccu
            ON tc.constraint_name = ccu.constraint_name
            AND tc.table_schema = ccu.table_schema
        WHERE tc.constraint_type = 'FOREIGN KEY'
        ORDER BY kcu.table_name, kcu.column_name
        """
        cursor = conn.cursor()
        cursor.execute(sql)
        return [
            {
                "constraint_name": row[0],
                "source_schema": row[1],
                "source_table": row[2],
                "source_column": row[3],
                "target_schema": row[4],
                "target_table": row[5],
                "target_column": row[6],
            }
            for row in cursor.fetchall()
        ]

    def get_indexes(self, conn, table_name: str, table_schema: str) -> list[dict]:
        sql = """
        SELECT
            i.relname AS index_name,
            array_agg(a.attname ORDER BY k.n) AS columns,
            ix.indisunique AS is_unique,
            ix.indisprimary AS is_primary
        FROM pg_index ix
        JOIN pg_class i ON i.oid = ix.indexrelid
        JOIN pg_class t ON t.oid = ix.indrelid
        JOIN pg_namespace n ON n.oid = t.relnamespace
        CROSS JOIN LATERAL unnest(ix.indkey) WITH ORDINALITY AS k(attnum, n)
        JOIN pg_attribute a ON a.attrelid = t.oid AND a.attnum = k.attnum
        WHERE t.relname = %s AND n.nspname = %s
        GROUP BY i.relname, ix.indisunique, ix.indisprimary
        ORDER BY i.relname
        """
        cursor = conn.cursor()
        cursor.execute(sql, (table_name, table_schema))
        return [
            {
                "name": row[0],
                "columns": list(row[1]) if row[1] else [],
                "is_unique": bool(row[2]),
                "is_primary": bool(row[3]),
            }
            for row in cursor.fetchall()
        ]

    def sample_rows(self, conn, table_name: str, table_schema: str, limit: int = 20) -> list[dict]:
        # 用参数化的 schema.table 格式（防注入用白名单校验）
        safe_name = f'"{table_schema}"."{table_name}"'
        sql = f"SELECT * FROM {safe_name} ORDER BY RANDOM() LIMIT {int(limit)}"
        cursor = conn.cursor()
        cursor.execute(sql)
        if not cursor.description:
            return []
        col_names = [desc[0].lower() for desc in cursor.description]
        return [dict(zip(col_names, row)) for row in cursor.fetchall()]

    def count_distinct(self, conn, table_name: str, table_schema: str, column_name: str) -> int:
        safe_table = f'"{table_schema}"."{table_name}"'
        sql = f'SELECT COUNT(DISTINCT "{column_name}") FROM {safe_table}'
        cursor = conn.cursor()
        cursor.execute(sql)
        return cursor.fetchall()[0][0]

    def get_enum_values(self, conn, table_name: str, table_schema: str, column_name: str, limit: int = 50) -> list[dict]:
        safe_table = f'"{table_schema}"."{table_name}"'
        sql = f'SELECT "{column_name}", COUNT(*) AS cnt FROM {safe_table} GROUP BY "{column_name}" ORDER BY cnt DESC LIMIT {int(limit)}'
        cursor = conn.cursor()
        cursor.execute(sql)
        return [
            {"value": str(row[0]) if row[0] is not None else None, "count": row[1]}
            for row in cursor.fetchall()
        ]

# apiscout/core/db_scanner/relations.py
"""关系推断 — 从字段命名模式推断表间关系"""
import re
import logging
from apiscout.core.db_scanner.models import InferredRelation

logger = logging.getLogger(__name__)

# 审计字段 → 用户表的映射
_AUDIT_COLUMNS = {"created_by", "updated_by", "modified_by", "deleted_by",
                  "create_by", "update_by", "creator", "updater"}

# 常见用户表名
_USER_TABLES = {"sys_user", "user", "users", "sys_users", "t_user", "t_users",
                "admin_user", "uc_user", "auth_user"}


def infer_relations(
    source_table: str,
    column_names: list[str],
    table_map: dict[str, dict],
) -> list[InferredRelation]:
    """从字段命名推断关系

    table_map: {"table_name": {"columns": {"col1", "col2", ...}}}
    """
    results = []
    all_tables = set(table_map.keys())

    for col in column_names:
        col_lower = col.lower()

        # 规则 1: parent_id → 同表自引用
        if col_lower == "parent_id" and "id" in table_map.get(source_table, {}).get("columns", set()):
            results.append(InferredRelation(
                source_table=source_table,
                source_column=col,
                target_table=source_table,
                target_column="id",
                confidence=0.85,
                evidence=f"自引用: {col} → {source_table}.id（树结构）",
            ))
            continue

        # 规则 2: created_by / updated_by → 用户表
        if col_lower in _AUDIT_COLUMNS:
            user_table = _find_user_table(all_tables)
            if user_table and "id" in table_map[user_table]["columns"]:
                results.append(InferredRelation(
                    source_table=source_table,
                    source_column=col,
                    target_table=user_table,
                    target_column="id",
                    confidence=0.8,
                    evidence=f"审计字段: {col} → {user_table}.id",
                ))
            continue

        # 规则 3: xxx_id → xxx.id
        match = re.match(r"^(.+)_id$", col_lower)
        if match:
            prefix = match.group(1)
            target = _find_table_by_prefix(prefix, all_tables)
            if target and target != source_table and "id" in table_map[target]["columns"]:
                results.append(InferredRelation(
                    source_table=source_table,
                    source_column=col,
                    target_table=target,
                    target_column="id",
                    confidence=0.9,
                    evidence=f"命名匹配: {col} → {target}.id",
                ))
            continue

        # 规则 4: xxx_code → xxx.code
        match = re.match(r"^(.+)_code$", col_lower)
        if match:
            prefix = match.group(1)
            target = _find_table_by_prefix(prefix, all_tables)
            if target and target != source_table and "code" in table_map[target]["columns"]:
                results.append(InferredRelation(
                    source_table=source_table,
                    source_column=col,
                    target_table=target,
                    target_column="code",
                    confidence=0.7,
                    evidence=f"编码匹配: {col} → {target}.code",
                ))

    return results


def _find_user_table(all_tables: set[str]) -> str | None:
    """在所有表中找用户表"""
    for candidate in _USER_TABLES:
        if candidate in all_tables:
            return candidate
    return None


def _find_table_by_prefix(prefix: str, all_tables: set[str]) -> str | None:
    """根据前缀找表名（支持 device_type_id → device_type）"""
    # 直接匹配
    if prefix in all_tables:
        return prefix
    # 加常见前缀 t_xxx, sys_xxx
    for p in ["t_", "sys_", "tb_"]:
        if f"{p}{prefix}" in all_tables:
            return f"{p}{prefix}"
    return None
```

# apiscout/core/db_scanner/sampler.py
"""枚举探测 + 随机采样"""
import logging
from copy import copy
from apiscout.core.db_scanner.models import ColumnInfo

logger = logging.getLogger(__name__)

# 枚举候选阈值
ENUM_CARDINALITY_THRESHOLD = 50

# 跳过枚举探测的类型
_SKIP_ENUM_TYPES = {"binary", "object"}


def scan_table_samples(
    conn,
    dialect,
    table_name: str,
    table_schema: str,
    columns: list[ColumnInfo],
    enum_threshold: int = ENUM_CARDINALITY_THRESHOLD,
    sample_limit: int = 20,
) -> tuple[list[ColumnInfo], list[dict]]:
    """对一张表做枚举探测 + 随机采样

    返回: (更新了 enum_values 的列列表, 样本行列表)
    """
    updated_columns = []

    for col in columns:
        new_col = copy(col)

        # 跳过主键、binary、object 类型
        if col.is_primary_key or col.normalized_type in _SKIP_ENUM_TYPES:
            updated_columns.append(new_col)
            continue

        # 枚举探测：先查基数
        try:
            cardinality = dialect.count_distinct(conn, table_name, table_schema, col.name)
            if 0 < cardinality < enum_threshold:
                enum_values = dialect.get_enum_values(conn, table_name, table_schema, col.name)
                new_col.enum_values = enum_values
                logger.debug("枚举候选: %s.%s (%d 个值)", table_name, col.name, cardinality)
        except Exception as e:
            logger.debug("枚举探测失败: %s.%s — %s", table_name, col.name, e)

        updated_columns.append(new_col)

    # 随机采样
    try:
        sample_rows = dialect.sample_rows(conn, table_name, table_schema, limit=sample_limit)
    except Exception as e:
        logger.warning("采样失败: %s — %s", table_name, e)
        sample_rows = []

    return updated_columns, sample_rows

# apiscout/core/db_scanner/connector.py
"""数据库连接管理 + 方言自动检测"""
import logging
from urllib.parse import urlparse

logger = logging.getLogger(__name__)

_DEFAULT_PORTS = {
    "postgresql": 5432,
    "mysql": 3306,
    "oracle": 1521,
    "mssql": 1433,
}

_DIALECT_ALIASES = {
    "postgres": "postgresql",
    "pg": "postgresql",
    "mariadb": "mysql",
    "sqlserver": "mssql",
}


def parse_connection_string(conn_str: str) -> dict:
    """解析连接字符串 → 结构化参数"""
    parsed = urlparse(conn_str)
    raw_dialect = parsed.scheme.split("+")[0].lower()
    dialect = _DIALECT_ALIASES.get(raw_dialect, raw_dialect)

    port = parsed.port or _DEFAULT_PORTS.get(dialect, 5432)

    return {
        "dialect": dialect,
        "host": parsed.hostname or "localhost",
        "port": port,
        "user": parsed.username or "",
        "password": parsed.password or "",
        "database": parsed.path.lstrip("/") if parsed.path else "",
    }


def get_dialect(dialect_name: str):
    """根据方言名获取 Dialect 实例"""
    if dialect_name == "postgresql":
        from apiscout.core.db_scanner.dialect.postgresql import PostgreSQLDialect
        return PostgreSQLDialect()
    elif dialect_name == "mysql":
        from apiscout.core.db_scanner.dialect.mysql import MySQLDialect
        return MySQLDialect()
    elif dialect_name == "oracle":
        from apiscout.core.db_scanner.dialect.oracle import OracleDialect
        return OracleDialect()
    elif dialect_name == "mssql":
        from apiscout.core.db_scanner.dialect.mssql import MSSQLDialect
        return MSSQLDialect()
    else:
        raise ValueError(f"不支持的数据库方言: {dialect_name}")


def connect(params: dict):
    """创建数据库连接，返回 (connection, dialect)"""
    dialect_name = params["dialect"]
    dialect = get_dialect(dialect_name)

    if dialect_name == "postgresql":
        import psycopg2
        conn = psycopg2.connect(
            host=params["host"], port=params["port"],
            user=params["user"], password=params["password"],
            dbname=params["database"],
        )
    elif dialect_name == "mysql":
        import mysql.connector
        conn = mysql.connector.connect(
            host=params["host"], port=params["port"],
            user=params["user"], password=params["password"],
            database=params["database"],
        )
    elif dialect_name == "oracle":
        import oracledb
        dsn = f"{params['host']}:{params['port']}/{params['database']}"
        conn = oracledb.connect(
            user=params["user"], password=params["password"], dsn=dsn,
        )
    elif dialect_name == "mssql":
        import pymssql
        conn = pymssql.connect(
            server=params["host"], port=params["port"],
            user=params["user"], password=params["password"],
            database=params["database"],
        )
    else:
        raise ValueError(f"不支持的数据库方言: {dialect_name}")

    logger.info("已连接: %s@%s:%d/%s", params["user"], params["host"], params["port"], params["database"])
    return conn, dialect

logger = logging.getLogger(__name__)


def _connect(conn_str: str = None, **kwargs):
    """内部连接函数（方便 mock）"""
    if conn_str:
        params = parse_connection_string(conn_str)
    else:
        params = kwargs
    return connect(params), params


def scan_database(
    conn_str: str = None,
    *,
    host: str = None,
    port: int = None,
    user: str = None,
    password: str = None,
    database: str = None,
    dialect: str = None,
) -> SchemaReport:
    """扫描数据库 schema，返回完整的 SchemaReport

    两种调用方式：
    1. scan_database("postgresql://user:pass@host/db")
    2. scan_database(host="...", port=5432, user="...", password="...", database="...", dialect="postgresql")
    """
    # 解析连接参数
    if conn_str:
        params = parse_connection_string(conn_str)
    else:
        params = {
            "dialect": dialect,
            "host": host,
            "port": port,
            "user": user,
            "password": password,
            "database": database,
        }

    db_name = params["database"]
    dialect_name = params["dialect"]

    conn, db_dialect = connect(params)

    try:
        # 1. 获取所有表
        raw_tables = db_dialect.get_tables(conn)
        logger.info("发现 %d 张表", len(raw_tables))

        # 2. 获取外键约束（全局一次查询）
        raw_fks = db_dialect.get_foreign_keys(conn)
        explicit_relations = [
            ExplicitRelation(
                source_table=fk["source_table"],
                source_column=fk["source_column"],
                target_table=fk["target_table"],
                target_column=fk["target_column"],
                constraint_name=fk["constraint_name"],
            )
            for fk in raw_fks
        ]
        logger.info("发现 %d 个外键约束", len(explicit_relations))

        # 3. 逐表扫描
        tables = []
        # 构建 table_map 供关系推断使用
        table_map = {}

        for raw_table in raw_tables:
            tname = raw_table["name"]
            tschema = raw_table["schema"]
            logger.info("扫描: %s.%s", tschema, tname)

            # 字段信息
            raw_columns = db_dialect.get_columns(conn, tname, tschema)
            columns = [
                ColumnInfo(
                    name=c["name"],
                    data_type=c["data_type"],
                    normalized_type=c["normalized_type"],
                    nullable=c["nullable"],
                    default=c["default"],
                    comment=c["comment"],
                    max_length=c["max_length"],
                    numeric_precision=c["numeric_precision"],
                    is_primary_key=c["is_primary_key"],
                    enum_values=None,
                )
                for c in raw_columns
            ]

            # 索引信息
            raw_indexes = db_dialect.get_indexes(conn, tname, tschema)
            indexes = [
                IndexInfo(
                    name=idx["name"],
                    columns=idx["columns"],
                    is_unique=idx["is_unique"],
                    is_primary=idx["is_primary"],
                )
                for idx in raw_indexes
            ]

            # 枚举探测 + 采样
            columns, sample_rows = scan_table_samples(
                conn, db_dialect, tname, tschema, columns,
            )

            table_info = TableInfo(
                name=tname,
                schema=tschema,
                comment=raw_table.get("comment"),
                row_count=raw_table.get("row_count"),
                columns=columns,
                indexes=indexes,
                sample_rows=sample_rows,
            )
            tables.append(table_info)

            # 更新 table_map
            table_map[tname] = {"columns": {c.name for c in columns}}

        # 4. 关系推断
        all_inferred = []
        # 排除已有显式外键的列
        explicit_pairs = {(r.source_table, r.source_column) for r in explicit_relations}
        for table in tables:
            cols_to_infer = [
                c.name for c in table.columns
                if (table.name, c.name) not in explicit_pairs
            ]
            inferred = infer_relations(table.name, cols_to_infer, table_map)
            all_inferred.extend(inferred)

        logger.info("推断 %d 个隐式关系", len(all_inferred))

        return SchemaReport(
            dialect=dialect_name,
            database=db_name,
            scanned_at=datetime.now(timezone.utc).isoformat(),
            tables=tables,
            explicit_relations=explicit_relations,
            inferred_relations=all_inferred,
        )

    finally:
        conn.close()
        logger.info("连接已关闭")
```

def scan_db(
    conn_str: str = None,
    output_dir: str = None,
    enrich_spec: str = None,
    **conn_kwargs,
) -> dict:
    """
    扫描数据库 schema → 生成报告。

    conn_str:     连接字符串（优先）
    output_dir:   输出目录（None 时自动用数据库名）
    enrich_spec:  已有 OpenAPI spec 路径（交叉增强）
    **conn_kwargs: host/port/user/password/database/dialect
    """
    import json
    from apiscout.core.db_scanner.introspector import scan_database
    from apiscout.core.db_scanner.connector import parse_connection_string
    from apiscout.core.generator.schema_report_html import generate_schema_report_html, write_schema_report_html
    from dataclasses import asdict

    # 扫描
    if conn_str:
        params = parse_connection_string(conn_str)
        report = scan_database(conn_str)
    else:
        params = conn_kwargs
        report = scan_database(**conn_kwargs)

    db_name = params.get("database", "unknown")

    # 输出目录
    if not output_dir:
        output_dir = f"./output/{db_name}"
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    # 写 JSON
    json_path = output_path / "schema_report.json"
    with open(json_path, "w", encoding="utf-8") as f:
        json.dump(asdict(report), f, ensure_ascii=False, indent=2, default=str)
    logger.info("生成: schema_report.json (%d 张表)", report.total_tables)

    # 写 HTML
    write_schema_report_html(report, str(output_path / "schema_report.html"))
    logger.info("生成: schema_report.html")

    # 交叉增强（可选）
    if enrich_spec:
        from apiscout.core.analyzer.schema_enricher import enrich_openapi_with_schema
        enriched = enrich_openapi_with_schema(enrich_spec, report)
        enriched_path = output_path / "enriched_spec.yaml"
        with open(enriched_path, "w", encoding="utf-8") as f:
            yaml.dump(enriched, f, default_flow_style=False, allow_unicode=True, sort_keys=False)
        logger.info("生成: enriched_spec.yaml（交叉增强）")

    return {
        "output_dir": str(output_path),
        "report": report,
    }

```python
# apiscout/core/analyzer/schema_enricher.py
"""用 DB schema 交叉增强 OpenAPI spec"""
import copy
import logging
import re

from apiscout.core.db_scanner.models import SchemaReport, ColumnInfo

logger = logging.getLogger(__name__)


def enrich_openapi_with_schema(
    spec: dict | str,
    schema_report: SchemaReport,
) -> dict:
    """用 DB schema 增强 OpenAPI spec

    匹配策略：API 路径段 → DB 表名，响应字段名 → DB 列名
    增强内容：enum、description、maxLength、nullable
    """
    import yaml

    if isinstance(spec, str):
        with open(spec, "r", encoding="utf-8") as f:
            spec_dict = yaml.safe_load(f)
    else:
        spec_dict = spec

    enriched = copy.deepcopy(spec_dict)

    # 构建 DB 列索引：table_name → {col_name → ColumnInfo}
    db_columns = {}
    for table in schema_report.tables:
        db_columns[table.name] = {col.name: col for col in table.columns}

    # 遍历所有路径
    for path, methods in enriched.get("paths", {}).items():
        # 从路径推断可能对应的表名
        table_name = _guess_table_from_path(path, set(db_columns.keys()))
        if not table_name:
            continue

        col_map = db_columns[table_name]
        logger.debug("匹配: %s → %s", path, table_name)

        # 遍历所有方法的 response schema
        for method, operation in methods.items():
            if not isinstance(operation, dict):
                continue
            _enrich_operation(operation, col_map)

    return enriched


def _guess_table_from_path(path: str, table_names: set[str]) -> str | None:
    """从 API 路径猜测对应的数据库表名

    /api/devices → device
    /api/v1/work-orders → work_order
    /api/work_orders/{id} → work_order
    """
    # 提取路径段（去掉参数占位符）
    segments = [s for s in path.split("/") if s and not s.startswith("{")]
    if not segments:
        return None

    # 从后往前试（最后一个有效段最可能是资源名）
    for seg in reversed(segments):
        # 规范化：devices → device, work-orders → work_order
        normalized = seg.lower().replace("-", "_")
        # 去复数
        if normalized.endswith("s") and not normalized.endswith("ss"):
            singular = normalized[:-1]
        else:
            singular = normalized

        # 直接匹配
        for candidate in [normalized, singular]:
            if candidate in table_names:
                return candidate
            # 试常见前缀
            for prefix in ["t_", "sys_", "tb_"]:
                if f"{prefix}{candidate}" in table_names:
                    return f"{prefix}{candidate}"

    return None


def _enrich_operation(operation: dict, col_map: dict[str, ColumnInfo]):
    """增强一个 operation 的 response schema"""
    for status_code, response in operation.get("responses", {}).items():
        if not isinstance(response, dict):
            continue
        for content_type, media in response.get("content", {}).items():
            if not isinstance(media, dict):
                continue
            schema = media.get("schema", {})
            _enrich_schema(schema, col_map)


def _enrich_schema(schema: dict, col_map: dict[str, ColumnInfo]):
    """递归增强 schema 中的属性"""
    if schema.get("type") == "array":
        items = schema.get("items", {})
        _enrich_schema(items, col_map)
        return

    if schema.get("type") == "object" or "properties" in schema:
        for prop_name, prop_schema in schema.get("properties", {}).items():
            if prop_name in col_map:
                col = col_map[prop_name]
                _apply_column_info(prop_schema, col)


def _apply_column_info(prop_schema: dict, col: ColumnInfo):
    """将 DB 列信息应用到 OpenAPI property schema"""
    # 补充 description（DB 注释）
    if col.comment and "description" not in prop_schema:
        prop_schema["description"] = col.comment

    # 补充 enum（枚举值）
    if col.enum_values and "enum" not in prop_schema:
        prop_schema["enum"] = [ev["value"] for ev in col.enum_values if ev["value"] is not None]

    # 补充 maxLength（字符类型）
    if col.max_length and col.normalized_type == "string" and "maxLength" not in prop_schema:
        prop_schema["maxLength"] = col.max_length

    # 补充 nullable
    if col.nullable and "nullable" not in prop_schema:
        prop_schema["nullable"] = True
```
